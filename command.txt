# ---- CLI Commands (After pip install decoyable) ----
# Show help
decoyable --help

# Scan for secrets only
decoyable scan secrets

# Scan for dependencies only  
decoyable scan deps

# Scan for SAST vulnerabilities
decoyable scan sast

# Scan for everything (comprehensive)
decoyable scan all

# Scan with custom path
decoyable scan all /path/to/scan

# Scan with verbose output
decoyable scan sast --format verbose

# Using the main module directly

# Start development server

# Start production server

# Start with SSL

# ---- API Testing ----
# Health check (VERIFIED WORKING)

# Test secrets scan endpoint

  -H "Content-Type: application/json" \
  -d '{"path": ".", "recursive": true}'

# Test dependencies scan endpoint

  -H "Content-Type: application/json" \
  -d '{"path": ".", "format": "json"}'

# Test SAST scan endpoint

  -H "Content-Type: application/json" \
  -d '{"path": ".", "output_format": "detailed"}'

# Test full scan endpoint
